class Solution:
    def findThePrefixCommonArray(self, A: List[int], B: List[int]) -> List[int]:
        n = len(A)
        ans = []
        pre_b = set()
        pre_a = set()
        inter = set()
        for i in range(n):
            pre_b.add(B[i])
            pre_a.add(A[i])
            if A[i] in pre_b:
                inter.add(A[i])
            if B[i] in pre_a:
                inter.add(B[i])
            ans.append(len(inter))
        return ans
```

class Solution:
    def findMaxFish(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        ans = 0
        for i in range(m):
            for j in range(n):
                if grid[i][j] > 0:
                    stack = [[i, j]]
                    cur = grid[i][j]
                    grid[i][j] = 0
                    while stack:
                        a, b = stack.pop()
                        for x, y in [[a + 1, b], [a, b + 1],[a, b - 1], [a - 1, b]]:
                            if 0 <= x < m and 0 <= y < n and grid[x][y] > 0:
                                stack.append([x, y])
                                cur += grid[x][y]
                                grid[x][y] = 0
                    ans = ans if ans > cur else cur
        return ans
```
